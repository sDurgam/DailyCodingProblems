	/*
	 * Given a list of numbers and a number k, return whether any two numbers from
	 * the list add up to k.
	 */

	static boolean isTargetSumPair(int[] arr, int k) {
		if (arr == null)
			return false;
		Arrays.sort(arr);
		int left = 0;
		int right = arr.length - 1;
		while (left < right) {
			int sum = arr[left] + arr[right];
			if (sum == k) {
				return true;
			} else if (sum < k) {
				left++;
			} else {
				right--;
			}
		}
		return false;
	}
